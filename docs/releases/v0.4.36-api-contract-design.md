### New API Types to Create

```typescript
// File: packages/types/src/api.ts

import type { JobStatus } from './job.js';
import type { ConflictType } from './assignment.js';

// ============================================================================
// Job API
// ============================================================================

/** POST /api/v1/jobs - Create job request */
export interface CreateJobRequest {
  reference: string;
  client: string;
  description: string;
  workshopExitDate: string; // YYYY-MM-DD
  status?: ApiJobStatus;
  tasksDsl?: string;         // Single-element jobs
  elements?: JcfElementInput[]; // Multi-element jobs
}

/** Element input for job creation */
export interface JcfElementInput {
  name: string;
  label?: string;
  sequence?: string;         // Task DSL
  prerequisiteNames?: string[];
}

/** Job status values used in API (lowercase, snake_case) */
export type ApiJobStatus =
  | 'draft'
  | 'planned'
  | 'in_progress'
  | 'delayed'
  | 'completed'
  | 'cancelled';

// ============================================================================
// Assignment API
// ============================================================================

/** POST/PUT /api/v1/tasks/{id}/assign - Assign/reschedule task request */
export interface AssignTaskRequest {
  targetId: string;           // Station or Provider UUID
  scheduledStart: string;     // ISO 8601
  bypassPrecedence?: boolean; // Alt-key bypass
}

/** Assignment operation response */
export interface AssignmentResponse {
  taskId: string;
  targetId: string;
  isOutsourced: boolean;
  scheduledStart: string;
  scheduledEnd: string | null;
  isCompleted: boolean;
  completedAt: string | null;
}

/** PUT /api/v1/tasks/{id}/completion - Completion toggle response */
export interface CompletionResponse {
  taskId: string;
  isCompleted: boolean;
  completedAt: string | null;
}

/** DELETE /api/v1/tasks/{id}/assign - Unassignment response */
export interface UnassignmentResponse {
  taskId: string;
  status: 'ready';
  message: string;
}

// ============================================================================
// Error Responses
// ============================================================================

/** 409 Conflict - Validation failed */
export interface ValidationErrorResponse {
  error: 'ValidationFailed';
  message: string;
  conflicts: ApiConflictDetail[];
  suggestedStart?: string;    // ISO 8601
}

/** Conflict detail in validation error */
export interface ApiConflictDetail {
  type: string;               // ConflictType from backend
  message: string;
  taskId: string;
  relatedTaskId?: string | null;
  targetId?: string | null;
  details?: Record<string, unknown> | null;
}

/** 404 Not Found */
export interface NotFoundErrorResponse {
  error: 'NotFound';
  message: string;
}

/** 400 Bad Request */
export interface InvalidRequestErrorResponse {
  error: 'InvalidRequest';
  message: string;
}

/** 503 Service Unavailable */
export interface ServiceUnavailableResponse {
  error: 'ServiceUnavailable';
  message: string;
}

/** Union of all API error responses */
export type ApiErrorResponse =
  | ValidationErrorResponse
  | NotFoundErrorResponse
  | InvalidRequestErrorResponse
  | ServiceUnavailableResponse;
```

### RTK Query Endpoint Design

```typescript
// Design document for scheduleApi

export const scheduleApi = createApi({
  reducerPath: 'scheduleApi',
  tagTypes: ['Snapshot', 'Job', 'ClientSuggestions'],

  endpoints: (builder) => ({
    // ========================================================================
    // Queries
    // ========================================================================

    getSnapshot: builder.query<ScheduleSnapshotResponse, void>({
      // M4: queryFn wraps mock
      // M5: query: () => '/schedule/snapshot'
      providesTags: ['Snapshot'],
    }),

    getClientSuggestions: builder.query<string[], string>({
      // query: (prefix) => `/jobs/clients?q=${prefix}`
      providesTags: ['ClientSuggestions'],
    }),

    lookupByReference: builder.query<{ client: string | null }, string>({
      // query: (ref) => `/jobs/lookup-by-reference?ref=${ref}`
    }),

    // ========================================================================
    // Mutations
    // ========================================================================

    createJob: builder.mutation<JobResponse, CreateJobRequest>({
      // query: (body) => ({ url: '/jobs', method: 'POST', body })
      invalidatesTags: ['Snapshot'],
    }),

    assignTask: builder.mutation<AssignmentResponse, { taskId: string; body: AssignTaskRequest }>({
      // query: ({ taskId, body }) => ({ url: `/tasks/${taskId}/assign`, method: 'POST', body })
      invalidatesTags: ['Snapshot'],
    }),

    rescheduleTask: builder.mutation<AssignmentResponse, { taskId: string; body: AssignTaskRequest }>({
      // query: ({ taskId, body }) => ({ url: `/tasks/${taskId}/assign`, method: 'PUT', body })
      invalidatesTags: ['Snapshot'],
    }),

    unassignTask: builder.mutation<void, string>({
      // query: (taskId) => ({ url: `/tasks/${taskId}/assign`, method: 'DELETE' })
      invalidatesTags: ['Snapshot'],
    }),

    toggleCompletion: builder.mutation<CompletionResponse, string>({
      // query: (taskId) => ({ url: `/tasks/${taskId}/completion`, method: 'PUT' })
      invalidatesTags: ['Snapshot'],
    }),
  }),
});
```

### Cache Invalidation Strategy

| Mutation | Invalidates | Reason |
|----------|-------------|--------|
| `createJob` | `['Snapshot']` | New job appears in snapshot |
| `assignTask` | `['Snapshot']` | Assignment added |
| `rescheduleTask` | `['Snapshot']` | Assignment updated |
| `unassignTask` | `['Snapshot']` | Assignment removed |
| `toggleCompletion` | `['Snapshot']` | Assignment state changed |

**Future optimization:** Use more granular tags like `['Job', jobId]` to avoid full snapshot refetch.
